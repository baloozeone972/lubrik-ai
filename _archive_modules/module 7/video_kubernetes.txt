# ============================================================================

# ============================================================================

# DEPLOYMENT - Service Java
apiVersion: apps/v1
kind: Deployment
metadata:
  name: video-service
  namespace: nexusai-production
  labels:
    app: video-service
    component: backend
    version: v1.0.0
spec:
  replicas: 3
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0
  selector:
    matchLabels:
      app: video-service
  template:
    metadata:
      labels:
        app: video-service
        component: backend
        version: v1.0.0
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8080"
        prometheus.io/path: "/actuator/prometheus"
    spec:
      # Sécurité
      securityContext:
        runAsNonRoot: true
        runAsUser: 1000
        fsGroup: 1000
      
      # Anti-affinity pour distribuer sur différents nodes
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
          - weight: 100
            podAffinityTerm:
              labelSelector:
                matchExpressions:
                - key: app
                  operator: In
                  values:
                  - video-service
              topologyKey: kubernetes.io/hostname
      
      containers:
      - name: video-service
        image: nexusai/video-service:1.0.0
        imagePullPolicy: IfNotPresent
        
        ports:
        - name: http
          containerPort: 8080
          protocol: TCP
        
        # Variables d'environnement depuis ConfigMap
        envFrom:
        - configMapRef:
            name: video-service-config
        - secretRef:
            name: video-service-secrets
        
        # Ressources
        resources:
          requests:
            memory: "1Gi"
            cpu: "500m"
          limits:
            memory: "2Gi"
            cpu: "1000m"
        
        # Health checks
        livenessProbe:
          httpGet:
            path: /actuator/health/liveness
            port: 8080
          initialDelaySeconds: 60
          periodSeconds: 10
          timeoutSeconds: 5
          failureThreshold: 3
        
        readinessProbe:
          httpGet:
            path: /actuator/health/readiness
            port: 8080
          initialDelaySeconds: 30
          periodSeconds: 5
          timeoutSeconds: 3
          failureThreshold: 3
        
        # Startup probe (pour démarrage lent)
        startupProbe:
          httpGet:
            path: /actuator/health
            port: 8080
          initialDelaySeconds: 0
          periodSeconds: 10
          timeoutSeconds: 3
          failureThreshold: 30
        
        # Volume mounts
        volumeMounts:
        - name: logs
          mountPath: /app/logs
      
      volumes:
      - name: logs
        emptyDir: {}

        value: 50
        periodSeconds: 60

      stabilizationWindowSeconds: 60
      policies:
      - type: Percent
        value: 100
        periodSeconds: 60

# DEPLOYMENT - Workers Python
apiVersion: apps/v1
kind: Deployment
metadata:
  name: video-worker
  namespace: nexusai-production
  labels:
    app: video-worker
    component: worker
    version: v1.0.0
spec:
  replicas: 3
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 2
      maxUnavailable: 1
  selector:
    matchLabels:
      app: video-worker
  template:
    metadata:
      labels:
        app: video-worker
        component: worker
        version: v1.0.0
    spec:
      # Tolérations pour nodes GPU (si disponibles)
      tolerations:
      - key: nvidia.com/gpu
        operator: Exists
        effect: NoSchedule
      
      # Node selector pour GPU (optionnel)
      nodeSelector:
        workload: video-generation
      
      containers:
      - name: video-worker
        image: nexusai/video-worker:1.0.0
        imagePullPolicy: IfNotPresent
        
        env:
        - name: KAFKA_BOOTSTRAP_SERVERS
          value: "kafka-service:9092"
        - name: KAFKA_TOPIC_REQUESTS
          value: "video.generation.requests"
        - name: KAFKA_TOPIC_EVENTS
          value: "video.generation.events"
        - name: AWS_REGION
          value: "eu-west-1"
        - name: S3_BUCKET_VIDEOS
          value: "nexusai-videos-prod"
        - name: S3_BUCKET_ASSETS
          value: "nexusai-video-assets-prod"
        - name: TEMP_DIR
          value: "/tmp/video-generation"
        - name: WORKER_ID
          valueFrom:
            fieldRef:
              fieldPath: metadata.name
        
        # Secrets
        - name: AWS_ACCESS_KEY_ID
          valueFrom:
            secretKeyRef:
              name: video-service-secrets
              key: AWS_ACCESS_KEY_ID
        - name: AWS_SECRET_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: video-service-secrets
              key: AWS_SECRET_ACCESS_KEY
        - name: OPENAI_API_KEY
          valueFrom:
            secretKeyRef:
              name: video-service-secrets
              key: OPENAI_API_KEY
        - name: ELEVENLABS_API_KEY
          valueFrom:
            secretKeyRef:
              name: video-service-secrets
              key: ELEVENLABS_API_KEY
        
        # Ressources (augmentées pour traitement vidéo)
        resources:
          requests:
            memory: "4Gi"
            cpu: "2000m"
            # ephemeral-storage: "20Gi"
          limits:
            memory: "8Gi"
            cpu: "4000m"
            # ephemeral-storage: "50Gi"
            # nvidia.com/gpu: 1  # Si GPU disponible
        
        # Volume pour fichiers temporaires
        volumeMounts:
        - name: temp-storage
          mountPath: /tmp/video-generation
      
      volumes:
      - name: temp-storage
        emptyDir:
          sizeLimit: 50Gi

# HPA - Auto-scaling des workers
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: video-worker-hpa
  namespace: nexusai-production
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: video-worker
  minReplicas: 3
  maxReplicas: 20
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 75
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80
  # Custom metric basé sur la taille de la queue Kafka
  - type: Pods
    pods:
      metric:
        name: kafka_consumer_lag
      target:
        type: AverageValue
        averageValue: "10"
  behavior:
    scaleDown:
      stabilizationWindowSeconds: 600  # 10 minutes
      policies:
      - type: Pods
        value: 2
        periodSeconds: 120
    scaleUp:
      stabilizationWindowSeconds: 120  # 2 minutes
      policies:
      - type: Pods
        value: 3
        periodSeconds: 60

# NETWORK POLICY - Sécurité réseau
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: video-service-netpol
  namespace: nexusai-production
spec:
  podSelector:
    matchLabels:
      app: video-service
  policyTypes:
  - Ingress
  - Egress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          name: ingress-nginx
    ports:
    - protocol: TCP
      port: 8080
  egress:
  # Autoriser vers PostgreSQL
  - to:
    - podSelector:
        matchLabels:
          app: postgres
    ports:
    - protocol: TCP
      port: 5432
  # Autoriser vers Redis
  - to:
    - podSelector:
        matchLabels:
          app: redis
    ports:
    - protocol: TCP
      port: 6379
  # Autoriser vers Kafka
  - to:
    - podSelector:
        matchLabels:
          app: kafka
    ports:
    - protocol: TCP
      port: 9092
  # Autoriser vers services externes (AWS, OpenAI)
  - to:
    - namespaceSelector: {}
    ports:
    - protocol: TCP
      port: 443

# CRONJOB - Nettoyage des vidéos échouées
apiVersion: batch/v1
kind: CronJob
metadata:
  name: video-cleanup-job
  namespace: nexusai-production
spec:
  schedule: "0 2 * * *"  # Tous les jours à 2h du matin
  successfulJobsHistoryLimit: 3
  failedJobsHistoryLimit: 3
  jobTemplate:
    spec:
      template:
        spec:
          restartPolicy: OnFailure
          containers:
          - name: cleanup
            image: nexusai/video-service:1.0.0
            command:
            - /bin/sh
            - -c
            - |
              psql $SPRING_DATASOURCE_URL -c "SELECT cleanup_failed_videos(30);"
            envFrom:
            - secretRef:
                name: video-service-secrets

# KEDA SCALEDOBJECT - Auto-scaling basé sur Kafka lag
apiVersion: keda.sh/v1alpha1
kind: ScaledObject
metadata:
  name: video-worker-kafka-scaler
  namespace: nexusai-production
spec:
  scaleTargetRef:
    name: video-worker
  minReplicaCount: 3
  maxReplicaCount: 20
  pollingInterval: 30
  cooldownPeriod: 300
  triggers:
  - type: kafka
    metadata:
      bootstrapServers: kafka-service:9092
      consumerGroup: video-generation-workers
      topic: video.generation.requests
      lagThreshold: "10"
      activationLagThreshold: "5"
